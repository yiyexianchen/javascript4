关于return返回值调用示例一：
function cost(num,value){ num =num+1; var total=num*value;

if(num>2&&num<3){
    total*=0.9;

}else if(num>=3&&num<5){
    total*=0.8;

}else if(num>6){
    total*=0.7;

}
return total;

}

var a=parseInt(prompt("请输入产品数量")); var b=parseFloat(prompt("请输入产品单价")); var all=cost(a,b); if(all<1000){ alert("你才花了"+all+"块钱"); alert("我们吃顿大餐吧"); }else{ alert("你花了"+all+"块钱"); alert("我们回家"); }
